# 设置最低CMake版本
cmake_minimum_required(VERSION 3.15)

# 定义项目名称和版本
project(SRastrizer VERSION 1.0)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# 定义源代码文件
set(SOURCE_FILES
    src/main.cpp
)

# 添加外部库路径
set(EXTERNAL_DIR ${CMAKE_SOURCE_DIR}/external)

# glad
add_library(glad STATIC ${EXTERNAL_DIR}/glad/src/glad.c)
target_include_directories(glad PUBLIC ${EXTERNAL_DIR}/glad/include)

# glfw precompiled
set(GLFW_INCLUDE_DIR ${EXTERNAL_DIR}/GLFW)
set(GLFW_LIBRARY ${EXTERNAL_DIR}/GLFW/glfw3.lib)

# 定义可执行文件
add_executable(softrasterizer ${SOURCE_FILES})

# 链接库
target_include_directories(softrasterizer PRIVATE src/common ${GLFW_INCLUDE_DIR})
target_link_libraries(softrasterizer PRIVATE glad ${GLFW_LIBRARY} ${CMAKE_DL_LIBS})
